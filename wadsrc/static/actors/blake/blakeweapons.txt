actor AutoChargePistol : WolfWeapon 314
{
	weapon.ammouse1 0
	+WEAPON.NOAUTOFIRE
	states
	{
		Ready:
			ACPG A 1 A_WeaponReady
			loop
		Fire:
			ACPG BC 3
			ACPG D 3 // Fire
			ACPG E 3
			goto Ready
		Spawn:
			ACPP A -1
			stop
	}
}
