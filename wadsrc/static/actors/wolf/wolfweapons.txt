actor WolfWeapon : Weapon
{
	weapon.ammotype1 "Clip"
	weapon.ammouse1 1
	weapon.ammogive1 6
	weapon.yadjust 19
	+WEAPON.NOGRIN
	states
	{
		Select:
			TNT1 A 0 A_Raise
			loop
		Deselect:
			TNT1 A 0 A_Lower
			loop
	}
}

actor Knife : WolfWeapon
{
	inventory.icon "KNIFE"
	weapon.ammouse1 0
	weapon.selectionorder 3700
	attacksound "weapon/knife/attack"
	+WEAPON.NOAUTOFIRE
	states
	{
		Ready:
			KNIF A 1 A_WeaponReady
			loop
		Fire:
			KNIF B 3
			KNIF C 3 A_CustomPunch(1, false, CPF_ALWAYSPLAYSOUND)
			KNIF DE 3
			goto Ready
	}
}
